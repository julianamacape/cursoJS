Eventos dentro do JavaScript

        
            <div id="area">
                Interaja
            </div>
            

            <script>
                var a = window.document.getElementById('area')
                a.addEventListener('click', quandoclicar)
                a.addEventListener('mouseenter', quandoentrar)
                a.addEventListener('mouseout', quandosair)

                function quandoclicar(){
                    a.innerText='Você clicou!'
                    a.style.background='black'
                    a.style.color='yellow'
                }

                function quandoentrar (){
                    a.innerText='Você entrou!'
                }

                function quandosair (){
                    a.innerText='Você saiu!'
                }
            </script>
